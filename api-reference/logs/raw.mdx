The Raw Logs API gives you direct access to the underlying event data that powers Profound's reports — including per-bot activity records and individual query logs. Use these endpoints when you need granular, row-level data for custom analytics, compliance auditing, or debugging anomalies in your aggregated report metrics.

Two methods are available: `bots` for bot-level activity events, and `logs` for individual query-level records.

## Bot Activity Logs

Retrieve raw bot activity events for your organization within a date range.

### Parameters

<ParamField query="start_date" type="string" required>
  The start of the log window in `YYYY-MM-DD` format.
</ParamField>

<ParamField query="end_date" type="string" required>
  The end of the log window in `YYYY-MM-DD` format.
</ParamField>

<ParamField query="bot_name" type="string">
  Filter logs to a specific AI bot.
</ParamField>

<ParamField query="model_id" type="string">
  Filter to a specific model version.
</ParamField>

<ParamField query="region_id" type="string">
  Restrict logs to a specific geographic region.
</ParamField>

<ParamField query="category_id" type="string">
  Filter to logs associated with a specific prompt category.
</ParamField>

<ParamField query="limit" type="integer">
  Maximum number of records to return per page.
</ParamField>

<ParamField query="cursor" type="string">
  Pagination cursor returned by a previous response. Pass this value to retrieve the next page of results.
</ParamField>

### Code Examples

<CodeGroup>

```bash curl
curl --request GET \
  --url "https://api.profound.ai/v1/logs/raw/bots?start_date=2025-01-01&end_date=2025-01-31" \
  --header "Authorization: Bearer $PROFOUND_API_KEY"
```

```typescript TypeScript
import { ProfoundClient } from "@profoundai/client";

const client = new ProfoundClient({ apiKey: process.env.PROFOUND_API_KEY });

const botLogs = await client.logs.raw.bots({
  start_date: "2025-01-01",
  end_date: "2025-01-31",
  bot_name: "chatgpt",
});

console.log(botLogs);
```

```python Python
import os
from profound import ProfoundClient

client = ProfoundClient(api_key=os.environ["PROFOUND_API_KEY"])

bot_logs = client.logs.raw.bots(
    start_date="2025-01-01",
    end_date="2025-01-31",
    bot_name="chatgpt",
)

print(bot_logs)
```

</CodeGroup>

## Query Logs

Retrieve individual query-level log records — one row per prompt execution — with full response metadata.

### Parameters

<ParamField query="start_date" type="string" required>
  The start of the log window in `YYYY-MM-DD` format.
</ParamField>

<ParamField query="end_date" type="string" required>
  The end of the log window in `YYYY-MM-DD` format.
</ParamField>

<ParamField query="bot_name" type="string">
  Filter to a specific bot.
</ParamField>

<ParamField query="category_id" type="string">
  Filter logs to a specific prompt category.
</ParamField>

<ParamField query="persona_id" type="string">
  Filter to logs run under a specific persona.
</ParamField>

<ParamField query="prompt_id" type="string">
  Filter to logs for a single tracked prompt.
</ParamField>

<ParamField query="region_id" type="string">
  Filter to a specific geographic region.
</ParamField>

<ParamField query="limit" type="integer">
  Maximum number of records to return per page.
</ParamField>

<ParamField query="cursor" type="string">
  Pagination cursor returned by a previous response. Pass this value to retrieve the next page of results.
</ParamField>

<Warning>
  Raw log endpoints return high-volume data. Use narrow date ranges and specific filters to avoid very large result sets. For bulk exports, paginate using the `cursor` returned in each response.
</Warning>

### Code Examples

<CodeGroup>

```bash curl

  --url "https://api.profound.ai/v1/logs/raw/logs?start_date=2025-01-01&end_date=2025-01-07&bot_name=gemini" \
```

### Paginating Through All Logs

<CodeGroup>

```typescript TypeScript
import { ProfoundClient } from "@profoundai/client";

const client = new ProfoundClient({ apiKey: process.env.PROFOUND_API_KEY });

const allLogs = [];
let cursor: string | undefined;

while (true) {
  const response = await client.logs.raw.logs({
    start_date: "2025-01-01",
    end_date: "2025-01-31",
    limit: 1000,
    cursor,
  });

  allLogs.push(...response.data);

  if (!response.meta?.cursor) break;
  cursor = response.meta.cursor;
}

console.log(`Fetched ${allLogs.length} log records.`);
```

```python Python
import os
from profound import ProfoundClient

client = ProfoundClient(api_key=os.environ["PROFOUND_API_KEY"])

all_logs = []
cursor = None

while True:
    response = client.logs.raw.logs(
        start_date="2025-01-01",
        end_date="2025-01-31",
        limit=1000,
        cursor=cursor,
    )

    all_logs.extend(response.data)

    cursor = response.meta.get("cursor") if response.meta else None
    if not cursor:
        break

print(f"Fetched {len(all_logs)} log records.")
```

</CodeGroup>

## Response Fields

### Bot Activity Logs Response

<ResponseField name="data" type="array">
  List of bot activity log records.

  <Expandable title="data[]">
    <ResponseField name="event_id" type="string">
      Unique identifier for this activity event.
    </ResponseField>

    <ResponseField name="bot_name" type="string">
      The AI bot involved in this activity event.
    </ResponseField>

    <ResponseField name="model_id" type="string">
      The model version, if available.
    </ResponseField>

    <ResponseField name="event_type" type="string">
      The type of bot activity recorded (e.g., `"query"`, `"crawl"`, `"citation"`).
    </ResponseField>

    <ResponseField name="timestamp" type="string">
      ISO 8601 timestamp of when the event occurred.
    </ResponseField>

    <ResponseField name="region_id" type="string">
      Geographic region associated with the event.
    </ResponseField>

    <ResponseField name="category_id" type="string">
      Prompt category associated with the event, if applicable.
    </ResponseField>

    <ResponseField name="metadata" type="object">
      Additional event-specific metadata.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata.

  <Expandable title="meta">
    <ResponseField name="total_records" type="integer">
      Total number of matching log records.
    </ResponseField>

    <ResponseField name="cursor" type="string">
      Cursor to pass as the `cursor` parameter to retrieve the next page of results. Absent when there are no further pages.
    </ResponseField>
  </Expandable>
</ResponseField>

### Query Logs Response

<ResponseField name="data" type="array">
  List of individual query log records.

  <Expandable title="data[]">
    <ResponseField name="log_id" type="string">
      Unique identifier for this query log record.
    </ResponseField>

    <ResponseField name="prompt_id" type="string">
      The Profound ID of the prompt that was executed.
    </ResponseField>

    <ResponseField name="prompt_text" type="string">
      The text of the prompt submitted to the AI assistant.
    </ResponseField>

    <ResponseField name="bot_name" type="string">
      The AI assistant that processed this query.
    </ResponseField>

    <ResponseField name="model_id" type="string">
      The model version used, if available.
    </ResponseField>

    <ResponseField name="response_text" type="string">
      The full text of the AI-generated response.
    </ResponseField>

    <ResponseField name="brand_mentioned" type="boolean">
      Whether your brand appeared in the response.
    </ResponseField>

    <ResponseField name="citation_count" type="integer">
      Number of URLs cited in the response.
    </ResponseField>

    <ResponseField name="timestamp" type="string">
      ISO 8601 timestamp of when the query was run.
    </ResponseField>

    <ResponseField name="region_id" type="string">
      Geographic region in which the query was executed.
    </ResponseField>

    <ResponseField name="category_id" type="string">
      Prompt category for this query.
    </ResponseField>

    <ResponseField name="persona_id" type="string">
      Persona used for this query, if applicable.
    </ResponseField>

    <ResponseField name="latency_ms" type="integer">
      Response latency in milliseconds.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata.

  <Expandable title="meta">
    <ResponseField name="total_records" type="integer">
      Total number of matching query log records.
    </ResponseField>

    <ResponseField name="cursor" type="string">
      Cursor to pass as the `cursor` parameter to retrieve the next page of results. Absent when there are no further pages.
    </ResponseField>
  </Expandable>
</ResponseField>
